# Editor tooling and task runner for Go development.

[install]
go-task.pkg-path = "go-task"
go-task.pkg-group = "go-tools"
gopls.pkg-path = "gopls"
gopls.pkg-group = "go-tools"
gopls.priority = 4
gotools.pkg-path = "gotools"
gotools.pkg-group = "go-tools"
gomodifytags.pkg-path = "gomodifytags"
gomodifytags.pkg-group = "go-tools"
gotests.pkg-path = "gotests"
gotests.pkg-group = "go-tools"
gore.pkg-path = "gore"
gore.pkg-group = "go-tools"
gum.pkg-path = "gum"
gum.pkg-group = "demo-tools"

[hook]
on-activate = '''
if [[ "${FLOX_ENVS_TESTING:-}" != "1" ]]; then
  gum style --border double --margin '1 2' \
    --padding '1 4' \
    'Go development environment' '' \
    'Included tools:' \
    '  go        - compiler and tools' \
    '  gopls     - language server' \
    '  gotools   - standard tools (goimports, etc.)' \
    '  go-task   - task runner' \
    '  gore      - REPL' '' \
    'Try the sample app:' \
    '  go build && ./hello'
fi
'''

Go Demo

Full Go development environment with editor tooling, a task runner, and a sample “Hello World” application.

Builds on the minimal go environment via [include].

Quick start

flox activate
go build && ./hello

Included tools

ToolDescription
goCompiler and standard toolchain
goplsOfficial Go language server
gotoolsStandard tools (goimports, etc.)
gomodifytagsStruct tag modifier
gotestsTest generator
goreGo REPL
go-taskTask runner (Taskfile.yml)
gumTerminal UI for scripts

Sample application

The included main.go prints a localized greeting using rsc.io/quote:

go build -o hello
./hello

Customization

Fork this environment and edit .flox/env/manifest.toml to add or remove tools. The [include] directive pulls in the base Go compiler from ../go.

Minimal environment

If you only need the Go compiler without editor tooling, use go directly or include it in your own manifest:

[include]
environments = ["flox/go"]
